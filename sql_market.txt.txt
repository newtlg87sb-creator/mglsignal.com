-- market_data хүснэгтийг үүсгэх
-- =========================================================
-- 2. MARKET DATA MANAGEMENT (Зах зээлийн өгөгдөл)
-- =========================================================

-- market_data хүснэгтийг үүсгэх
create table IF NOT EXISTS market_data (
  symbol text primary key,
  full_symbol text,
  bid float8,
  ask float8,
  spread float8,
  min_usdt float8,
  volume float8,
  real_change float8,
  h1_change float8,
  change_24h float8,
  is_st boolean default false,
  min_order_cost float8,
  updated_at timestamptz default now()
);

-- Дутуу багана нэмэх хамгаалалт (Хүснэгт аль хэдийн үүссэн үед хэрэгтэй)
ALTER TABLE market_data ADD COLUMN IF NOT EXISTS min_order_cost float8;
ALTER TABLE market_data ADD COLUMN IF NOT EXISTS is_st boolean DEFAULT false;

-- Realtime-ийг идэвхжүүлэх (Сайт дээр шууд харагдахад хэрэгтэй)
-- Хэрэв аль хэдийн нэмэгдсэн бол алгасах логик
DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_publication_tables 
    WHERE pubname = 'supabase_realtime' 
    AND schemaname = 'public' 
    AND tablename = 'market_data'
  ) THEN
    ALTER PUBLICATION supabase_realtime ADD TABLE market_data;
  END IF;
END $$;

-- Фронтенд дээр өгөгдөл унших эрх нэмэх (Select Policy)
DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_policy 
    WHERE polname = 'Enable read access for all users' 
    AND polrelid = 'market_data'::regclass
  ) THEN
    ALTER TABLE market_data ENABLE ROW LEVEL SECURITY;
    CREATE POLICY "Enable read access for all users" ON market_data FOR SELECT USING (true);
  END IF;
END $$;
